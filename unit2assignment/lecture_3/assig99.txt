#include <stdio.h>
int main ()
{
	int i  ,n;
	float avg=0 , num,sum=0 ;
	printf("Enter the max number of inputs :");
	fflush(stdin); fflush(stdout);
	scanf("%d" , &n);
	for(i = 0 ; i<n  ;i++)
	{
		printf("Enter n%d : ",i+1);
		fflush(stdin); fflush(stdout);
		scanf("%f" , &num);
		if(num<0)
			break;
		sum += num;

	}
	avg = sum/n ;
	printf("the avg = %f" , avg) ;
	return 0 ;
}
