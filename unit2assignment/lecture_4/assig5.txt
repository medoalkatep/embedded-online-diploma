#include <stdio.h>
int main ()
{
	int n , i , num , loc ;
	printf("Enter no of elements : ");
	fflush(stdin);  fflush(stdout);
	scanf("%d",&n);
	int x[n];
	for(i = 0 ; i<n ; i++)
	{
		scanf("%d",&x[i]);
	}
	printf("Enter element to be searched : ");
	fflush(stdin);  fflush(stdout);
	scanf("%d",&num);
	for(i = 0 ; i < n ; i++)
	{
		if (x[i]== num){
			loc= i+1 ;
			break ;
		}
	}
	printf("number found at the location : %d" , loc);

	return 0 ;
}
